    <script>
        function navigate(destination) {
            const feedback = document.getElementById('feedback');
            const title = document.getElementById('feedback-title');
            const desc = document.getElementById('feedback-desc');
            const progress = document.getElementById('progress');
            
            // Reset animation
            progress.style.animation = 'none';
            progress.offsetHeight; // Trigger reflow
            progress.style.animation = 'loading 3s ease-in-out forwards';
            
            title.textContent = `Navigating to ${destination}...`;
            desc.textContent = 'Robot is calculating optimal route';
            
            feedback.classList.add('visible');
            
            // Update current location after "navigation"
            setTimeout(() => {
                document.querySelector('.current-location strong').textContent = destination;
                title.textContent = `Arrived at ${destination}`;
                desc.textContent = 'Robot has reached destination';
            }, 3000);
        }

        function emergencyStop() {
            const feedback = document.getElementById('feedback');
            const title = document.getElementById('feedback-title');
            const desc = document.getElementById('feedback-desc');
            const progress = document.getElementById('progress');
            
            progress.style.animation = 'none';
            progress.style.width = '100%';
            progress.style.background = '#dc2626';
            
            title.textContent = 'Emergency Stop Activated';
            desc.textContent = 'Robot has stopped all operations';
            
            feedback.classList.add('visible');
        }
    </script>
